### **1 & 19. JS value â†’ JSON string (basic stringify)**

```js
function myStringify(val) {
  if (val === null) return "null";
  if (typeof val === "number" || typeof val === "boolean") return String(val);
  if (typeof val === "string") return `"${val}"`;
  if (Array.isArray(val)) return `[${val.map(myStringify)}]`;
  if (typeof val === "object") {
    return `{${Object.entries(val).map(([k,v]) => `"${k}":${myStringify(v)}`)}}`;
  }
}
```

### **2. Deep copy with circular refs**

```js
function deepClone(obj, map = new WeakMap()) {
  if (obj === null || typeof obj !== "object") return obj;
  if (map.has(obj)) return map.get(obj);
  const copy = Array.isArray(obj) ? [] : {};
  map.set(obj, copy);
  for (let k in obj) copy[k] = deepClone(obj[k], map);
  return copy;
}
```

### **5. Insertion Sort**

```js
function insertionSort(arr) {
  for (let i = 1; i < arr.length; i++) {
    let key = arr[i], j = i - 1;
    while (j >= 0 && arr[j] > key) arr[j + 1] = arr[j--];
    arr[j + 1] = key;
  }
  return arr;
}
```

### **6. Memoize (multi-arg)**

```js
function memoize(fn) {
  const cache = new Map();
  return (...args) => {
    const key = JSON.stringify(args);
    if (!cache.has(key)) cache.set(key, fn(...args));
    return cache.get(key);
  };
}
```

### **10. Recursive transform**

```js
function transform(obj, fn) {
  if (Array.isArray(obj)) return obj.map(v => transform(v, fn));
  if (obj && typeof obj === "object") {
    return Object.fromEntries(Object.entries(obj).map(([k,v]) => [k, transform(v, fn)]));
  }
  return fn(obj);
}
```

### **11. Deep equal**

```js
function deepEqual(a, b) {
  if (a === b) return true;
  if (typeof a !== "object" || typeof b !== "object" || !a || !b) return false;
  const keys = Object.keys(a);
  if (keys.length !== Object.keys(b).length) return false;
  return keys.every(k => deepEqual(a[k], b[k]));
}
```

### **13. Squash object (flatten)**

```js
function squash(obj) {
  const result = {};

  function helper(current, prefix) {
    for (const key of Object.keys(current)) {
      const value = current[key];
      const newKey = prefix ? `${prefix}.${key}` : key;

      if (
        value !== null &&
        typeof value === "object" &&
        !Array.isArray(value) &&
        !(value instanceof Date)
      ) {
        helper(value, newKey); // recurse
      } else {
        result[newKey] = value; // assign leaf
      }
    }
  }

  helper(obj, "");
  return result;
}

```

### **21. Debounce with cancel**

function debounce(fn, d) {
  let t;
  const f = (...a) => { clearTimeout(t); t = setTimeout(() => fn(...a), d); };

}
