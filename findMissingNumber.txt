  arr.sort((a, b) => a - b);
  console.log("arr", arr);
  let first = arr[0],
    last = arr[arr.length - 1];
  let count = arr.length + 1;
  let diff = (last - first) / (count - 1);
  let currentSum = arr.reduce((acc, num) => acc + num, 0);
  let Sum = count/2 * (first + last);

  return Sum - currentSum;
}
